------------ DB/EntityFramework ----------------

install entity framework core from nugget
install sqlite 
add to appsettings.dev.json file
  "ConnectionStrings": {
    "DefaultConnection": "Data source=pinkpapertrail.db"
  }

  create class that inherrits from dbcontext; create dbset

  add dbcontext to startup.cs 
  services.AddDbContext<StoreContext>(options =>
                    options.UseSqlite(_config
                    .GetConnectionString("DefaultConnection")));
         

dotnet tool list -g
dotnet tool install --global dotnet-ef --version 5.0.4
dotnet ef migrations add InitialCreate -o  Data/Migrations

dotnet ef database -g
dotnet ef database update
